  function chessBoard() {
        var size = prompt("Please enter a Number");
        var line = '';
        var print = "";
        var i = 0;

        for (i = 0; i < size / 2; i++) {
          line = print + line + '# ';
        }

        for (i = 0; i < size; i++) {
          if (i % 2 === 0) {
            print = print + (' ' + line) + "
";
          } else {
            print = print + (line) + "
";
          }
        }
        $("#q3result").html(print);
      }

        chessBoard();
      });

            
              function countBs(bString) {
        var count = 0;
        for (var i = 0; i < bString.length; i++) {
          if (bString.charAt(i) == 'B') {
            count = (count + 1);
          }
        }
        return count;
      }


      $("#q4button").on("click", function() {

        var userInput = prompt("");
        var x = countBs(userInput);

        $("#q4result").html(x);
      });
